# イラク、周辺地域を併合
news_event = {
	id = my_news.1
	title = my_news.1.t
	desc = my_news.1.desc
	picture = GFX_news_event_soldiers_marching

	major = yes

	is_triggered_only = yes

	option = {
		name = my_news.1.a
		trigger = {
			NOT = {
				TAG = AFG
				TAG = IRQ
				TAG = PER
				TAG = TUR
			}
		}
	}
	option = {
		name = my_news.1.b
		trigger = { TAG = IRQ }
	}
	option = {
		name = my_news.1.c
		trigger = {
			OR = {
				TAG = AFG
				TAG = PER
				TAG = TUR
			}
		}
	}
}
